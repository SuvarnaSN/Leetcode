# Write your MySQL query statement below

(SELECT u.name as results
FROM Users as u JOIN (SELECT user_id, COUNT(*) as count
FROM MovieRating
GROUP BY user_id) AS v
ON u.user_id = v.user_id
ORDER BY v.count DESC, u.name ASC   # if multiple users have same count order by name lexicographically = ASC
LIMIT 1)
UNION
(SELECT m.title as results
FROM MovieRating AS r JOIN Movies as m
ON r.movie_id = m.movie_id
WHERE YEAR(created_at) = 2020 AND MONTH(created_at) = 02
GROUP BY m.movie_id    # group by is necessary becayse we need to find average for every movie
order by AVG(r.rating) DESC, m.title ASC
LIMIT 1)
