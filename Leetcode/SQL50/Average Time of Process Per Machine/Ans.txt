SELECT machine_id, ROUND(AVG(timestamp-prev_time), 3)  AS  processing_time 
FROM ( 

    SELECT machine_id, process_id, activity_type, timestamp, 
    LAG(timestamp)  OVER (   
        PARTITION BY machine_id, process_id    
        ORDER BY timestamp   
    ) AS prev_time           
    FROM Activity            

)t            
WHERE activity_type = 'end'    
GROUP BY machine_id;
